# A Test

This is a `inline code block`: `if not A, return 0`.

class Solution:
    """
    @param A: an integer rotated sorted array
    @param target: an integer to be searched
    @return: an integer
    """
    def search(self, A, target):
        # write your code here
        
        if not A:
            return -1
        
        last = A[-1]
        start, end = 0, len(A) - 1
        
        if target > last:
            # On the left, larger sequence
            while start + 1 < end:
                mid = (start + end) // 2
                if A[mid] == target:
                    return mid
                elif A[mid] < target:
                    if A[mid] > last:
                        start = mid
                    else:
                        end = mid
                else:
                    end = mid
        
        else:
            # On the right, smaller sequence
            while start + 1 < end:
                mid = (start + end) // 2
                if A[mid] == target:
                    return mid
                elif A[mid] < target:
                    start = mid
                else:
                    if A[mid] > last:
                        start = mid
                    else:
                        end = mid
        
        if A[start] == target:
            return start
        elif A[end] == target:
            return end
        else:
            return -1
